[docs]def set_jobid(path, jobid):
    fin = open(path, "r")
    with open(path + "_bak", "w") as fout:
        for line in fin:
            if line.find("jobid") == -1:
                fout.write(line)
            else:
                fout.write(line.replace("jobid", jobid))




[docs]def run_hetero_sampler_host_test(jobid):
    set_jobid("./hetero_sampler_host_test.py", jobid)
    cmd = ["python", "./hetero_sampler_host_test.py_bak"]
    process = subprocess.Popen(cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
    stdout, stderr = process.communicate()
    stderr = stderr.decode("utf-8")
    print (stderr)
    process.wait()
    os.system("rm hetero_sampler_host_test.py_bak")
