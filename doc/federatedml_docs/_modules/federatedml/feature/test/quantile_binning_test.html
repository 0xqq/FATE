        quan_bin = QuantileBinning(bin_param)
        t0 = time.time()

        t1 = time.time()
        print("Split points fitting time: {}".format(t1 - t0))
        for col_name, col_idx in self.col_dict.items():

            for bin_idx, percent in enumerate(bin_percent):
