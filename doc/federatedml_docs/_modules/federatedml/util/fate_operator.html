def _one_dimension_dot(X, w):
    res = 0
    for i in range(len(X)):
        if np.fabs(X[i]) < 1e-5:
            continue
        res += w[i] * X[i]
    return res

[docs]def reduce_add(x, y):
    if x is None and y is None:
        return None

    if x is None:
        return y

    if y is None:
        return x
    if not isinstance(x, Iterable):
        result = x + y
    else:
        result = []
        for idx, acc in enumerate(x):
            result.append(acc + y[idx])
    return result
